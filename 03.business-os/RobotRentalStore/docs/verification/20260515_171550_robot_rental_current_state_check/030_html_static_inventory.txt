    async function confirmContract() {
      if (!savedRentalContractId) {
        alert("先に見積を保存してください。");
        return;
      }

      const ok = confirm("申込を確定します。よろしいですか？");
      if (!ok) return;

      try {
        const res = await fetch(`${API_BASE}/api/v1/business/robot-rental/contracts/confirm`, {
          method: "POST",
          headers: { "content-type": "application/json", ...authHeaders() },
          body: JSON.stringify({ rental_contract_id: savedRentalContractId })
        });

        const json = await res.json();
        if (!json.ok) throw new Error(json.error || "申込確定に失敗しました");

        renderMessage(`
          <strong>申込を確定しました</strong><br>
          申込番号: ${json.rental_contract_id || "-"}<br>
          状態: ${json.contract_status || "confirmed"}<br>
          次の手続き: 決済へ進む
        `);
      } catch (err) {
        renderMessage(`申込確定失敗: ${err.message}`, false);
      }
    }

    async function createPaymentIntent() {
      if (!savedRentalContractId) {
        alert("先に見積保存と申込確定を行ってください。");
        return;
      }

      const ok = confirm("決済予定を作成します。よろしいですか？");
      if (!ok) return;

      try {
        const res = await fetch(`${API_BASE}/api/v1/business/robot-rental/payments/intent/create`, {
          method: "POST",
          headers: { "content-type": "application/json", ...authHeaders() },
          body: JSON.stringify({ rental_contract_id: savedRentalContractId })
        });

        const json = await res.json();
        if (!json.ok) throw new Error(json.error || "決済予定の作成に失敗しました");

        renderMessage(`
          <strong>決済予定を作成しました</strong><br>
          決済予定番号: ${json.rental_payment_intent_id || "-"}<br>
          金額: ¥${Number(json.amount_jpy || 0).toLocaleString()}<br>
          状態: ${json.payment_status || "pending"}
        `);
      } catch (err) {
        renderMessage(`決済予定作成失敗: ${err.message}`, false);
      }
    }

    async function startRental() {
      if (!savedRentalContractId) {
        alert("先に見積保存、申込確定、決済予定作成を行ってください。");
        return;
      }

      const ok = confirm("利用を開始します。よろしいですか？");
      if (!ok) return;

      try {
        const res = await fetch(`${API_BASE}/api/v1/business/robot-rental/rentals/start`, {
          method: "POST",
          headers: { "content-type": "application/json", ...authHeaders() },
          body: JSON.stringify({ rental_contract_id: savedRentalContractId })
        });

        const json = await res.json();
        if (!json.ok) throw new Error(json.error || "利用開始に失敗しました");

        renderMessage(`
          <strong>利用開始しました</strong><br>
          契約番号: ${json.rental_contract_id || "-"}<br>
          状態: ${json.contract_status || "active"}<br>
          レンタル状態: ${json.period_status || "active"}<br>
          残り秒数: ${json.remaining_seconds_snapshot ?? "-"}
        `);
      } catch (err) {
        renderMessage(`利用開始失敗: ${err.message}`, false);
      }
    }
