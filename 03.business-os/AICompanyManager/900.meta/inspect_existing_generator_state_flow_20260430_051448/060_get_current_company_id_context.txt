  function parseCsvLine(line) {
    var cells = [];
    var current = "";
    var inQuotes = false;
    var i;
    var ch;
    var next;

    for (i = 0; i < line.length; i += 1) {
      ch = line.charAt(i);
      next = line.charAt(i + 1);

      if (ch === '"' && inQuotes && next === '"') {
        current += '"';
        i += 1;
      } else if (ch === '"') {
        inQuotes = !inQuotes;
      } else if (ch === "," && !inQuotes) {
        cells.push(current);
        current = "";
      } else {
        current += ch;
      }
    }

    cells.push(current);
    return cells.map(function (cell) {
      return cell.trim();
    });
